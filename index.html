    <script>

        /**

         * La función `calculateResults` es responsable de realizar los cálculos de la estrategia de inversión

         * y actualizar el HTML con los resultados. Utiliza los valores de entrada del usuario para determinar

         * el capital final tanto para la estrategia como para una simple inversión "naked".

         */

        const calculateResults = () => {

            // Obtener valores de los campos de entrada

            const initialCapital = parseFloat(document.getElementById('initialCapital').value);

            const gainFactor = parseFloat(document.getElementById('gainFactor').value);

            const annualGridGain = parseFloat(document.getElementById('annualGridGain').value);

            const years = parseFloat(document.getElementById('years').value);


            // Obtener los elementos HTML para mostrar los resultados

            const resultsDiv = document.getElementById('results');

            const detailedResultsDiv = document.getElementById('detailedResults');

            const finalStrategyCapitalElem = document.getElementById('finalStrategyCapital');

            const finalNakedCapitalElem = document.getElementById('finalNakedCapital');

            const conclusionTextElem = document.getElementById('conclusionText');


            // Validación de entrada para asegurar que se pueda realizar un cálculo adecuado

            if (isNaN(initialCapital) || isNaN(gainFactor) || isNaN(annualGridGain) || isNaN(years) || initialCapital <= 0 || gainFactor <= 0 || annualGridGain < 0 || years < 1) {

                // Mostrar mensaje de error si las entradas no son válidas

                resultsDiv.style.display = 'block';

                detailedResultsDiv.style.display = 'none';

                finalStrategyCapitalElem.textContent = 'Error';

                finalNakedCapitalElem.textContent = 'Error';

                conclusionTextElem.textContent = 'Por favor, ingrese valores válidos en todos los campos.';

                return;

            }


            // Calcular el capital inicial para el bot y el margen de seguridad

            // basándose en la relación proporcionada

            const botCapitalRatio = 10000 / 43800;

            const initialBotCapital = initialCapital * botCapitalRatio;

            const initialMarginCapital = initialCapital - initialBotCapital;


            // Actualizar el texto de desglose del capital inicial dinámicamente

            document.getElementById('initialCapitalBreakdown').innerHTML = `El capital inicial de <strong>$${initialCapital.toLocaleString('es-MX', { maximumFractionDigits: 0 })}</strong> se distribuye de la siguiente manera: $${initialBotCapital.toLocaleString('es-MX', { maximumFractionDigits: 0 })} se asignan al bot (capital de trabajo) y $${initialMarginCapital.toLocaleString('es-MX', { maximumFractionDigits: 0 })} se reservan como margen de seguridad.`;


            // Paso 1: Calcular los porcentajes de ganancia por apreciación de precio y trading de rejilla

            // El apalancamiento 3x se aplica a la apreciación del precio (gainFactor - 1)

            const appreciationGainPercent = (gainFactor - 1) * 300;

            // La ganancia de la rejilla es un porcentaje anual constante multiplicado por el número de años

            const gridGainPercent = annualGridGain * years;

            const totalGainPercent = appreciationGainPercent + gridGainPercent;


            // Paso 2: Calcular el número total de "eventos de ganancia"

            // Cada evento es una ganancia del 5% sobre el capital del bot

            const totalEvents = Math.floor(totalGainPercent / 5);


            // Paso 3: Usar la fórmula de interés compuesto para el capital del bot

            // 1/5 de cada evento de ganancia del 5% se reinvierte, lo que es una tasa de interés compuesto del 1% por evento

            const compoundingRatePerEvent = 0.01;

            const finalBotCapital = initialBotCapital * Math.pow(1 + compoundingRatePerEvent, totalEvents);


            // Paso 4: Calcular la ganancia total del bot y la cantidad transferida al margen

            const totalBotGain = finalBotCapital - initialBotCapital;

            // La cantidad transferida es 4 veces la ganancia total del bot, ya que 4/5 de la ganancia va al margen

            const transferredToMargin = totalBotGain * 4;

            const finalMarginCapital = initialMarginCapital + transferredToMargin;


            // Paso 5: Resumen Final

            const finalStrategyCapital = finalBotCapital + finalMarginCapital;

            const nakedCapital = initialCapital * gainFactor;


            // Mostrar los resultados principales

            finalStrategyCapitalElem.textContent = `$${finalStrategyCapital.toLocaleString('es-MX', { maximumFractionDigits: 0 })}`;

            finalNakedCapitalElem.textContent = `$${nakedCapital.toLocaleString('es-MX', { maximumFractionDigits: 0 })}`;


            let conclusion = "";

            if (finalStrategyCapital > nakedCapital) {

                const percentageSuperiority = ((finalStrategyCapital / nakedCapital) - 1) * 100;

                conclusion = `¡Esta estrategia supera a la inversión directa por un ${percentageSuperiority.toLocaleString('es-MX', { maximumFractionDigits: 0 })}%!`;

            } else {

                const percentageSuperiority = ((nakedCapital / finalStrategyCapital) - 1) * 100;

                conclusion = `La inversión directa supera a la estrategia por un ${percentageSuperiority.toLocaleString('es-MX', { maximumFractionDigits: 0 })}%.`;

            }

            conclusionTextElem.textContent = conclusion;


            // Mostrar las secciones de resultados

            resultsDiv.style.display = 'block';

            detailedResultsDiv.style.display = 'block';


            // Mostrar el desglose detallado

            document.getElementById('yearsText').textContent = years;

            document.getElementById('appreciationGainPercent').textContent = `${appreciationGainPercent.toLocaleString('es-MX', { maximumFractionDigits: 0 })}%`;

            document.getElementById('gridGainPercent').textContent = `${gridGainPercent.toLocaleString('es-MX', { maximumFractionDigits: 0 })}%`;

            document.getElementById('totalStrategyGainPercent').textContent = `${totalGainPercent.toLocaleString('es-MX', { maximumFractionDigits: 0 })}%`;

            document.getElementById('totalEvents').textContent = `${totalEvents} eventos`;

            document.getElementById('finalBotCapital').textContent = `$${finalBotCapital.toLocaleString('es-MX', { maximumFractionDigits: 0 })}`;

            document.getElementById('finalMarginCapital').textContent = `$${finalMarginCapital.toLocaleString('es-MX', { maximumFractionDigits: 0 })}`;

            document.getElementById('finalStrategyCapitalDetail').textContent = `$${finalStrategyCapital.toLocaleString('es-MX', { maximumFractionDigits: 0 })}`;

        };
